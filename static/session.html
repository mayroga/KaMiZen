"en-US" : "es-ES"; msg.rate = 0.95; window.speechSynthesis.cancel(); window.speechSynthesis.speak(msg); } catch (e) { console.warn("Error en salida de voz:", e); } } /* ========================= MOTOR DE RENDERIZADO ========================= */ function render() { const app = document.getElementById("app"); if (!app) return; app.innerHTML = ""; /* --- BIENVENIDA --- */ if (state.step === "welcome") { app.innerHTML = `

AL CIELO

Intelligence, Awareness, Action, Truth

`; return; } /* --- CONFIRMACIÓN --- */ if (state.step === "named") { app.innerHTML = `

Protocol Ready: ${state.name}

The journey to awareness begins now.

`; return; } /* --- MOTOR DE HISTORIAS --- */ if (state.step === "story") { const story = state.stories[state.sIndex]; if (!story) { state.step = "missions"; state.mIndex = 0; render(); return; } const text = state.lang === "en" ? story.en : story.es; speak(text); app.innerHTML = `

Knowledge Base: ${story.t}

${text}

Knowledge Unit ${state.sIndex + 1} / ${state.stories.length}

`; return; } /* --- MOTOR DE MISIONES (EXTRACCIÓN DE BLOQUES b) --- */ if (state.step === "missions") { const mission = state.missions[state.mIndex]; if (!mission) { state.step = "complete"; render(); return; } // Buscamos el bloque 'd' dentro de 'b' para obtener pregunta y opciones const dataBlock = mission.b.find(block => block.t === "d"); const question = dataBlock ? (state.lang === "en" ? dataBlock.q.en : dataBlock.q.es) : "..."; const options = dataBlock ? dataBlock.op : []; const correctIdx = dataBlock ? dataBlock.c : 0; app.innerHTML = `

Mission ${mission.id}: ${mission.cat.toUpperCase()}

${question}

${options.map((opt, i) => `
${opt}
`).join("")}

Mission ${state.mIndex + 1} / ${state.missions.length}

`; return; } /* --- FINALIZACIÓN Y RESPIRACIÓN --- */ if (state.step === "complete") { app.innerHTML = `

Training Phase Complete

Inhale

This exercise trains calm focus and nervous system control.

`; startBreathing(); return; } } /* ========================= CONTROLES DE FLUJO ========================= */ function startApp() { const input = document.getElementById("nameInput"); state.name = input?.value?.trim() || "User"; state.step = "named"; render(); } function startStories() { state.step = "story"; state.sIndex = 0; render(); } function nextStory() { state.sIndex++; render(); } function nextMission() { state.mIndex++; render(); } function checkAnswer(index, correct) { const fb = document.getElementById("feedback"); if (!fb) return; if (index === correct) { fb.innerHTML = `

CORRECT: Awareness increases safety.

REVISE: Analyze the risk again.

`; speak("Analyze the risk again."); } } function startBreathing() { let inhale = true; const interval = setInterval(() => { const text = document.getElementById("breathText"); if (!text) { clearInterval(interval); return; } text.innerText = inhale ? "Inhale" : "Exhale"; inhale = !inhale; }, 4000); } function clearState() { if(confirm("Delete current session data?")) { restart(); } } function restart() { state.step = "welcome"; state.sIndex = 0; state.mIndex = 0; state.name = ""; render(); }
